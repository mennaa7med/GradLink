   ─────────────────
   doc = fitz.open(pdf_path)
   text = ""
   for page in doc:
       text += page.get_text()
   doc.close()

   الجديدة (PyPDF2):
   ─────────────────
   text = ""
   with open(pdf_path, "rb") as f:
       reader = PyPDF2.PdfReader(f)
       for page in reader.pages:
           text += page.extract_text()
